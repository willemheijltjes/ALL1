\newcommand{\speechbubbleright}[3][1]{%
  \newdimen\witdim
  \witdim=#2
  \newdimen\heidim
  \heidim=#1\baselineskip \advance\heidim by 1.2ex
  \includegraphics[height=\heidim,width=1.3\witdim]{frogs/SprechblaseNeuFlip.png}%
  \hskip-1.045\witdim%
  \raise1.2ex\hbox{\parbox[b]{\witdim}{#3\strut}}%
  \hskip.045\witdim%
}
  
\newcommand{\speechbubbleleft}[3][1]{%
  \newdimen\witdim
  \witdim=#2
  \newdimen\heidim
  \heidim=#1\baselineskip \advance\heidim by 1.2ex
  \includegraphics[height=\heidim,width=1.3\witdim]{frogs/SprechblaseNeu.png}%
  \hskip-1.245\witdim%
  \raise1.2ex\hbox{\parbox[b]{\witdim}{#3\strut}}%
  \hskip.245\witdim%
}
  
\newcommand\speakgreen[3][1]{%
  \newdimen\risdim
  \risdim=\ifcase#1 2.5ex\or 2.5ex \or 1.5ex\or .5ex\or 0ex \or -1ex 
  \or -3.5ex \or -4.5ex \else -8.5ex \fi
  %%\ifcase#3 2.5 \or 1.5\or .5\or 0 \else -1 \fi xx #3
  \bgfrog\!%
  \raise\risdim\hbox{\speechbubbleright[#1]{#2}{#3}}}

\newcommand\speakorange[3][1]{%
  \newdimen\risdim
  \risdim=\ifcase#1 2.5ex\or 2.5ex \or 1.5ex\or .5ex\or 0ex \or 
  -1ex \or -3.5ex\else -6.5ex \fi
  \raise\risdim\hbox{\speechbubbleleft[#1]{#2}{#3}}%
  \!\bofrog%
}

\newcommand\speakviolet[3][1]{%
  \newdimen\risdim
  \risdim=\ifcase#1 2.5ex\or 2.5ex \or 1.5ex\or .5ex\or 0ex \or 
  -1ex \or -3.5ex\else -6.5ex \fi
  \raise\risdim\hbox{\speechbubbleleft[#1]{#2}{#3}}%
  \!\bvfrog%
}

\newcommand\speakingobjectleft[5]{%
  \raise#1\hbox{\speechbubbleleft[#2]{#3}{#4}}%
  #5%
}
  
\newcommand\speakingobjectright[5]{%
  \raise#1\hbox{\speechbubbleright[#2]{#3}{#4}}%
  #5%
}
